// Script để cài đặt Telegram Bot Properties
// Chạy 1 lần duy nhất trong Apps Script > Run > setupTelegramProperties

function setupTelegramProperties() {
  const props = PropertiesService.getScriptProperties();
  
  // === CẤU HÌNH ===
  const config = {
    TELEGRAM_BOT_TOKEN: '',
    TELEGRAM_DRIVE_FOLDER_ID: '',
    TELEGRAM_SHEET_ID: '',
    WEB_APP_CLEAR_CACHE_URL: ''
  };
  
  // Lưu tất cả config
  Object.keys(config).forEach(key => {
    props.setProperty(key, config[key]);
  });
  
  console.log('Đã cài đặt Telegram Properties:');
  console.log(JSON.stringify(config, null, 2));
  
  return config;
}

// Chạy để reset properties
function resetTelegramProperties() {
  PropertiesService.getScriptProperties().deleteAllProperties();
  console.log('Đã xóa tất cả properties');
}

// Kiểm tra properties hiện tại
function checkTelegramProperties() {
  const props = PropertiesService.getScriptProperties().getProperties();
  console.log('Properties hiện tại:', JSON.stringify(props, null, 2));
  return props;
}
