// DO GET - API ENDPOINTS
// ===========================
function doGet(e) {
  const mode = e?.parameter?.mode || '';
  const cache = CacheService.getScriptCache();
  const ss = SpreadsheetApp.openById(CONFIG.WEB_SHEET_ID);

  try {
    switch (mode) {
      case 'products':
        const cachedProducts = cache.get(CONFIG.CACHE_KEY_PRODUCTS);
        if (cachedProducts) {
          return ContentService
            .createTextOutput(cachedProducts)
            .setMimeType(ContentService.MimeType.JSON);
        }
        const products = getProductsFromSheet(ss);
        const productsJson = JSON.stringify(products);
        cache.put(CONFIG.CACHE_KEY_PRODUCTS, productsJson, CONFIG.CACHE_TTL_PRODUCTS);
        return ContentService
          .createTextOutput(productsJson)
          .setMimeType(ContentService.MimeType.JSON);

      case 'banners':
        const cachedBanners = cache.get(CONFIG.CACHE_KEY_BANNER);
        if (cachedBanners) {
          return ContentService
            .createTextOutput(cachedBanners)
            .setMimeType(ContentService.MimeType.JSON);
        }
        const banners = getImagesFromFolder(CONFIG.FOLDER_ID_BANNER);
        const bannersJson = JSON.stringify(banners);
        cache.put(CONFIG.CACHE_KEY_BANNER, bannersJson, 21600);
        return ContentService
          .createTextOutput(bannersJson)
          .setMimeType(ContentService.MimeType.JSON);

      case 'customers':
        const cachedCustomers = cache.get(CONFIG.CACHE_KEY_CUSTOMER);
        if (cachedCustomers) {
          return ContentService
            .createTextOutput(cachedCustomers)
            .setMimeType(ContentService.MimeType.JSON);
        }
        const customers = getCustomersFromSheet(ss);
        const customersJson = JSON.stringify(customers);
        cache.put(CONFIG.CACHE_KEY_CUSTOMER, customersJson, 21600);
        return ContentService
          .createTextOutput(customersJson)
          .setMimeType(ContentService.MimeType.JSON);

      case 'clearCache':
        cache.remove(CONFIG.CACHE_KEY_PRODUCTS);
        return ContentService
          .createTextOutput(JSON.stringify({ status: 'cleared' }))
          .setMimeType(ContentService.MimeType.JSON);

      default:
        return ContentService
          .createTextOutput(JSON.stringify({
            status: 'OK',
            message: 'i10 Store API',
            endpoints: ['?mode=products', '?mode=banners', '?mode=customers']
          }))
          .setMimeType(ContentService.MimeType.JSON);
    }
  } catch (err) {
    Logger.log('doGet ERROR: ' + err);
    return ContentService
      .createTextOutput(JSON.stringify({ error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// ===========================
// DO POST - HANDLE SUBMISSIONS
// ===========================
function doPost(e) {
  try {
    if (!e?.postData?.contents) {
      return jsonError('No data received');
    }

    const data = JSON.parse(e.postData.contents);
    const timestamp = new Date();
    const ss = SpreadsheetApp.openById(CONFIG.WEB_SHEET_ID);

    if (data.soHD && data.tenA && data.tenB && data.model) {
      return handleContract(data, timestamp, ss);
    }
    if (data.soHD && data.tenKhachHang && data.products) {
      return handleInvoice(data, timestamp, ss);
    }
    if (data.product && data.name && data.phone) {
      return handleOrder(data, timestamp, ss);
    }
    if (data.message && data.name && data.phone) {
      return handleContact(data, timestamp, ss);
    }

    return jsonError('Invalid data format');
  } catch (err) {
    Logger.log('doPost ERROR: ' + err);
    return jsonError(err.message);
  }
}

// INVOICE HANDLER
// ===========================
function handleInvoice(data, timestamp, ss) {
  const sheetName = CONFIG.HOADON_SHEET_NAME;
  let sheet = ss.getSheetByName(sheetName);

  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    const headers = [
      'Thời gian tạo', 'Số hóa đơn', 'Ngày tháng năm',
      'Tên khách hàng', 'Địa chỉ khách hàng', 'SĐT khách hàng',
      'Tên sản phẩm', 'Model/Serial', 'Cấu hình', 'Số lượng',
      'Đơn giá', 'Thành tiền', 'Giá tiền (chữ)', 'Thanh toán', 'Ghi chú'
    ];
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    styleHeaderRow(sheet, 1, headers.length);
  }

  const products = data.products || [];
  let thanhTien = 0;

  products.forEach(p => {
    const sl = parseInt(p.soLuong) || 0;
    const dg = parseInt(String(p.donGia).replace(/[^0-9]/g, '')) || 0;
    thanhTien += sl * dg;
  });

  const rows = products.map((p, index) => [
    timestamp.toLocaleString('vi-VN'),
    data.soHD || '',
    data.ngayThangNam || '',
    data.tenKhachHang || '',
    data.diaChiKhachHang || '',
    data.sdtKhachHang || '',
    p.tenSanPham || '',
    p.modelSanPham || '',
    p.cauHinh || '',
    p.soLuong || '',
    p.donGia || '',
    (p.soLuong && p.donGia) ? (parseInt(p.soLuong) * parseInt(String(p.donGia).replace(/[^0-9]/g, ''))) : '',
    index === 0 ? data.giaTienChu : '',
    index === 0 ? data.thanhToan : '',
    index === 0 ? data.ghiChu : ''
  ]);

  if (rows.length > 0) {
    const startRow = sheet.getLastRow() + 1;
    sheet.getRange(startRow, 1, rows.length, rows[0].length).setValues(rows);
    for (let i = 0; i < rows.length; i++) {
      styleDataRow(sheet, startRow + i, rows[i].length);
    }
  }

  try {
    const userEmail = Session.getActiveUser().getEmail();
    if (userEmail && userEmail.includes('@')) {
      MailApp.sendEmail({
        to: userEmail,
        subject: '🧾 Hóa đơn: ' + (data.tenKhachHang || 'N/A'),
        body: `Hóa đơn bán hàng vừa được tạo:\n\n` +
          `Số HĐ: ${data.soHD || 'N/A'}\n` +
          `Ngày: ${data.ngayThangNam || 'N/A'}\n` +
          `Khách: ${data.tenKhachHang || 'N/A'}\n` +
          `SĐT: ${data.sdtKhachHang || 'N/A'}\n` +
          `Sản phẩm: ${products[0]?.tenSanPham || 'N/A'}\n` +
          `Tổng: ${thanhTien.toLocaleString('vi-VN')} VNĐ (${data.giaTienChu || 'N/A'})\n` +
          `Thanh toán: ${data.thanhToan || 'N/A'}\n` +
          `Thời gian: ${timestamp.toLocaleString('vi-VN')}`
      });
    }
  } catch (err) {
    Logger.log('Email skipped: ' + err.message);
  }

  return jsonSuccess('Đã lưu hóa đơn!');
}

function getTelegramImageMap(ss) {
  const sheet = ss.getSheetByName(CONFIG.TELEGRAM_IMAGES_SHEET);
  if (!sheet) return {};

  const data = sheet.getDataRange().getValues();
  let map = {};

  for (let i = 1; i < data.length; i++) {
    const serial = (data[i][1] || '').toString().trim();
    const images = data[i][2];

    if (!serial || !images) continue;

    try {
      map[serial] = JSON.parse(images);
    } catch (e) {}
  }

  return map;
}

function getProductsFromSheet(ss) {
  const telegramMap = getTelegramImageMap(ss);
  const sheet = ss.getSheetByName(CONFIG.WEB_SHEET_NAME);
  if (!sheet) throw new Error('Sheet "Web" không tồn tại');

  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return [];

  const headers = data[0];
  const rows = data.slice(1);

  return rows.map(row => {
    const obj = {};
    headers.forEach((h, i) => {
      obj[h] = row[i];
    });

    if (obj['Web Link']) {
      obj.slug = obj['Web Link'];
    } else {
      const slugText = [obj['Model'] || obj['Name'], obj['CPU'], obj['RAM'], obj['GPU']]
        .filter(Boolean)
        .join(' ');
      obj.slug = 'san-pham/' + createSlug(slugText || 'product');
    }

    const serial =
      obj['ID'] ||
      (obj['S/N'] ? obj['S/N'].toString().slice(-3) : '');

    const folderId = extractFolderId(obj['Folder Album Image Google Drive']);

    let images = [];

    if (telegramMap[serial]) {
      images = telegramMap[serial];
    } else if (folderId) {
      images = getImagesFromFolder(folderId);
    } else if (obj['images']) {
      try {
        images = typeof obj['images'] === 'string'
          ? JSON.parse(obj['images'])
          : obj['images'];
      } catch (e) {
        images = [];
      }
    }

    obj.images = images;
    obj.photoLink = obj['Photos'] || obj['Photo'] || '';

    return obj;
  }).filter(p => p.Brand);
}

function getCustomersFromSheet(ss) {

// DRIVE HELPERS
// ===========================
function getImagesFromFolder(folderId) {
  if (!folderId) return [];
  try {
    const q = `'${folderId}' in parents and mimeType contains 'image/' and trashed = false`;
    const res = Drive.Files.list({ q, fields: 'files(id, name, mimeType)', orderBy: 'name' });
    return (res.files || []).map(f => ({
      id: f.id,
      name: f.name,
      url: `https://drive.google.com/uc?export=view&id=${f.id}`,
      thumb: `https://drive.google.com/thumbnail?id=${f.id}&sz=s1000`
    }));
  } catch (err) {
    Logger.log('Drive error: ' + err);
    return [];
  }
}

// ===========================
// FOLDER UTILITIES
// ===========================
function extractFolderId(url) {
  if (!url || typeof url !== 'string') return '';
  const m1 = url.match(/\/folders\/([a-zA-Z0-9_-]{20,})/);
  if (m1) return m1[1];
  const m2 = url.match(/[?&]id=([a-zA-Z0-9_-]{20,})/);
  if (m2) return m2[1];
  const m3 = url.match(/\b([a-zA-Z0-9_-]{25,44})\b/);
  if (m3) return m3[1];
  return '';
}

// UI HELPERS
// ===========================
function createSlug(text) {
  if (!text) return '';
  return text.toString().toLowerCase()
    .replace(/\s+/g, '-')
    .replace(/[^\w\-]+/g, '')
    .replace(/\-\-+/g, '-')
    .replace(/^-+/, '')
    .replace(/-+$/, '');
}

function saveJsonToDrive(jsonContent, fileName) {
  try {
    const folder = DriveApp.getFolderById(CONFIG.STATIC_JSON_FOLDER_ID);
    const files = folder.getFilesByName(fileName);
    while (files.hasNext()) {
      files.next().setTrashed(true);
    }
    const file = folder.createFile(fileName, jsonContent, 'application/json');
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return file.getId();
  } catch (e) {
    Logger.log('saveJson ERROR: ' + e);
    throw e;
  }
}

// Check empty folders with batch processing
function checkEmptyImageFolders() {
  const ss = SpreadsheetApp.openById(CONFIG.WEB_SHEET_ID);
  const sheet = ss.getSheetByName(CONFIG.WEB_SHEET_NAME);
  if (!sheet) {
    SpreadsheetApp.getUi().alert('Không tìm thấy sheet "Web"');
    return;
  }

  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const nameCol = headers.indexOf('Name');
  const driveCol = headers.indexOf('Folder Album Image Google Drive');

  if (nameCol === -1 || driveCol === -1) {
    SpreadsheetApp.getUi().alert('Thiếu cột Name hoặc Folder Album Image Google Drive');
    return;
  }

  const emptyFolders = [];
  sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn()).setBackground(null);

  const folderChecks = [];
  for (let i = 1; i < data.length; i++) {
    const name = (data[i][nameCol] || '').toString().trim();
    const driveLink = (data[i][driveCol] || '').toString().trim();
    if (!driveLink || !name) continue;
    const folderId = extractFolderId(driveLink);
    if (!folderId) continue;
    folderChecks.push({ row: i + 1, name, link: driveLink, folderId });
  }

  const BATCH_SIZE = 20;
  for (let batchStart = 0; batchStart < folderChecks.length; batchStart += BATCH_SIZE) {
    const batch = folderChecks.slice(batchStart, batchStart + BATCH_SIZE);
    batch.forEach(item => {
      try {
        const query = "'" + item.folderId + "' in parents and mimeType contains 'image/' and trashed = false";
        const files = Drive.Files.list({ q: query, fields: 'files(id)' }).files || [];
        if (files.length === 0) {
          emptyFolders.push({ row: item.row, name: item.name, link: item.link });
          sheet.getRange(item.row, 1, 1, sheet.getLastColumn()).setBackground('#fff8b3');
        }
      } catch (err) {
        emptyFolders.push({ row: item.row, name: item.name, link: item.link, error: err.message });
        sheet.getRange(item.row, 1, 1, sheet.getLastColumn()).setBackground('#ffcccc');
      }
    });
    if (batchStart + BATCH_SIZE < folderChecks.length) {
      Utilities.sleep(200);
    }
  }

  if (emptyFolders.length === 0) {
    SpreadsheetApp.getActive().toast('✅ Tất cả folder đều có ảnh');
  } else {
    const html = `<h3>Phát hiện ${emptyFolders.length} folder rỗng</h3>
      <table border="1" style="width:100%">
        <tr><th>Hàng</th><th>Tên</th><th>Link</th><th>Lỗi</th></tr>
        ${emptyFolders.map(f =>
      `<tr><td>${f.row}</td><td>${f.name}</td><td><a href="${f.link}" target="_blank">Mở</a></td><td>${f.error || 'Không có ảnh'}</td></tr>`
    ).join('')}
      </table>`;
    SpreadsheetApp.getUi().showSidebar(HtmlService.createHtmlOutput(html).setTitle('Folder rỗng'));
  }
}

function styleHeaderRow(sheet, row, cols) {
  const range = sheet.getRange(row, 1, 1, cols);
  range.setFontWeight('bold');
  range.setBackground('#2c3e50');
  range.setFontColor('#ffffff');
  range.setHorizontalAlignment('center');
  range.setVerticalAlignment('middle');
  sheet.setFrozenRows(row);
}
function styleDataRow(sheet, row, cols) {
  const range = sheet.getRange(row, 1, 1, cols);
  range.setBorder(true, true, true, true, true, true);
  if (row % 2 === 0) range.setBackground('#f0f8ff');
}
